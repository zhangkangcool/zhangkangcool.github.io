Example1

class X {
public:
  double i;
};

class A: public virtual X {
public:
  double j;
};

class B: public virtual X {
public:
  double k;
};

class C: public A, public B {
  double l;
};

void foo() {
  C c;
}

Example2

class X {
public:
  double i;
  virtual void fooX();
};

class A: public virtual X {
public:
  double j;
  virtual void fooA();
};

class B: public virtual X {
public:
  double k;
  virtual void fooB();
};

class C: public A, public B {
  double l;
  virtual void fooC();
};

void foo() {
  C c;
}
