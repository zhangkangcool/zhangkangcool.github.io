1. 得到本地时间

#include <stdio.h>
#include <sys/time.h>
#include <time.h>

int main() {
    struct timeval tv;
    struct tm *tm_info;

    // 获取当前时间
    if (gettimeofday(&tv, NULL) == -1) {
        perror("gettimeofday");
        return 1;
    }

    // 将秒数转换为本地时间
    tm_info = localtime(&tv.tv_sec);

    // 输出当前的时分秒
    printf("当前时间（时:分:秒）: %02d:%02d:%02d\n", tm_info->tm_hour, tm_info->tm_min, tm_info->tm_sec);

    return 0;
}
