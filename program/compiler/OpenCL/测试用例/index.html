void* getFileContent(char *filename, size_t *length)
{
    char* text;
    FILE *pf = fopen(filename, "r");
    fseek(pf, 0, SEEK_END);
    long lSize = ftell(pf);
    text = (char*)malloc(lSize+1);
    rewind(pf);
    fread(text, sizeof(char), lSize, pf);
    text[lSize] = '\0';
    *length = lSize;
    return text;
}
